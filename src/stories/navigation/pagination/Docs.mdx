```tsx

      size="size-sm"
      spacing="tight"
      totalPages={10}
      variant="outlined"

        label={faAnglesLeft}
        onClick={() => {}}
        disabled={false}
    />
    <PaginationArrow
        id="prev"

        indexEnd={10}
        totalAmount={100}
    />

        label={faChevronRight}
        onClick={() => {}}
        disabled={false}
    />
    <PaginationArrow
        id="last"
        label={faAnglesRight}
        onClick={() => {}}
```
